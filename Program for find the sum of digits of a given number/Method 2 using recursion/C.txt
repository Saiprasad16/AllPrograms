// C program to find the sum of digits of a number

# include<stdio.h>

int sum_of_digits(int n)
{
    return n == 0 ? 0 : n%10 + sum_of_digits(n/10) ;
}

int main()
{
    int n;
    printf(“\nEnter a number : “);
    scanf(“%d”,&n);
    printf(“\nSum of digits of %d is %d\n”, n,sum_of_digits(n));
    return 0;
}
