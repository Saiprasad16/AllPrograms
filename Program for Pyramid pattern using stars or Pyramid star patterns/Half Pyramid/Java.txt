// Java program to print half pyramid pattern using stars
import java.util.*;
public class Main
{
    public static void main(String[] args)
    {
        int n, i, j;
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();
        for(i = 0; i < n; i++)
        {
            for(j = 0; j <= i; j++)
            {
                System.out.print(“*”);
            }
            System.out.print(“\n”);
        }
    }
}
