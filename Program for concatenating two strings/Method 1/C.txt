// C program to concatenate two strings

#include
int main()
{
    char str1[100], str2[100], i, j;
    printf(“\nEnter first string: “);
    scanf(“%s”, str1);
    printf(“\nEnter second string: “);
    scanf(“%s”, str2);
    for(i = 0; str1[i] != ‘\0’; ++i);

    for(j = 0; str2[j] != ‘\0’; ++j, ++i)
    {
        str1[i] = str2[j];
    }
    str1[i] = ‘\0’;
    printf(“Resultant string : %s”, str1);
    return 0;
}
