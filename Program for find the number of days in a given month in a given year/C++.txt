#include<iostream>
using namespace std;

{

    int year, month;
    cin >> month >> year;
    if(month == 1 || month == 3 || month == 5 || month == 7 || month == 8 || month == 10 || month == 12)
        cout << “Number of days is 31”;
    else if((month == 2) && (year%4 == 0) || ((year%100 == 0) &&(year%400 == 0)))
    {
        cout << “Number of days is 29”;
    }
    else if(month == 2)
    {
        cout << “Number of days is 28”;
    }

}
