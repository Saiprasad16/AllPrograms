// Java program to find the number of times digit 3 occurs in each and every number from 0 to n

import java.util.*;

public class Main
{
    static int count_3s(int n)
    {
        int count = 0;
        while (n > 0)
        {
            if (n % 10 == 3)
            {
                count++;
            }
            n = n / 10;
        }
        return count;
    }

    static int count_in_range(int n)
    {
        int count = 0 ;
        for (int i = 2; i <= n; i++)
        {
            count += count_3s(i);
        }
        return count;
    }

    {

        number = sc.nextInt();
        System.out.print(count_in_range(number));
    }
}
