/* C++ program to remove spaces from a string */
#include <iostream>
using namespace std;

char *remove_white_spaces(char *str)
{
    int i = 0, j = 0;
    while (str[i])
    {
        if (str[i] != ‘ ‘)
            str[j++] = str[i];
        i++;
    }
    str[j] = ‘\0’;
    return str;
}

int main()
{
    char str[50];
    cout << “\nEnter a string : “;
    cin.getline(str,50);
    remove_white_spaces(str);
    cout << “Resultant String : “;
    cout << str << endl;
    return 0;
}
