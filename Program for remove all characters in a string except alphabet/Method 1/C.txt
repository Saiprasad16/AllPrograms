// C program to remove all Characters in a String Except Alphabet

#include

int main()
{
    char input[150];
    int i, j;
    printf(“Enter a string : “);
    gets(input);

    for(i = 0; input[i] != ‘\0’; ++i)
    {
        while (!( (input[i] >= ‘a’ && input[i] <= ‘z’) || (input[i] >= ‘A’ && input[i] <= ‘Z’) || input[i] == ‘\0’) )
        {
            for(j = i; input[j] != ‘\0’; ++j)
            {
                input[j] = input[j+1];
            }
            input[j] = ‘\0’; 
        }
    }
    printf(“\nOutput string : “);
    puts(input);
    printf(“\n”);
    return 0;
}
